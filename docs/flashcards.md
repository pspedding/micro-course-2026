let idx = 0;
let flipped = false;

(function() {
  const modules = [...new Set(ALL_CARDS.map(c => c.module))].sort();
  const sel = document.getElementById('module-filter');
  modules.forEach(m => {
    const opt = document.createElement('option');
    opt.value = m; opt.textContent = m;
    sel.appendChild(opt);
  });
})();

function filterCards() {
  const mod = document.getElementById('module-filter').value;
  const typ = document.getElementById('type-filter').value;
  deck = ALL_CARDS.filter(c => (mod==='ALL'||c.module===mod) && (typ==='ALL'||c.type===typ));
  idx = 0; flipped = false; showCard();
}

function shuffleDeck() {
  for (let i=deck.length-1;i>0;i--) { const j=Math.floor(Math.random()*(i+1)); [deck[i],deck[j]]=[deck[j],deck[i]]; }
  idx=0; flipped=false; showCard();
}

function resetDeck() { filterCards(); }

  if (deck.length===0) { document.getElementById('card-content').innerHTML='<em>No cards match.</em>'; return; }

  const el = document.getElementById('card');
  document.getElementById('card-content').innerHTML = flipped ? card.back : card.front;
  el.style.background = flipped ? '#e0f7fa' : 'white';
  el.style.borderColor = flipped ? '#00695c' : '#00897b';
  document.getElementById('flip-hint').textContent = flipped ? '👆 Tap to see question' : '👆 Tap to reveal answer';
  document.getElementById('counter').textContent = `Card ${idx+1} of ${deck.length}`;
  document.getElementById('progress-bar').style.width = `${((idx+1)/deck.length)*100}%`;
  document.getElementById('module-badge').textContent = `${card.module} · ${card.lesson} · ${card.topic.replace(/-/g,' ')}`;
}
